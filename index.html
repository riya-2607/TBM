// ── INIT ─────────────────────────────────────────────────────────────────────

(async function init() {

  try {

    msalInstance = new msal.PublicClientApplication(MSAL_CONFIG);

    await msalInstance.initialize();

    const accounts = msalInstance.getAllAccounts();

    if (accounts.length > 0) {

      currentAccount = accounts[0];

      showSignedIn(currentAccount);

    }

  } catch(err) {

    console.warn("MSAL init:", err.message);

  }

  setToday();

})();


function setToday() {

  const t = new Date();

  document.getElementById("date").value =

    `${t.getFullYear()}-${String(t.getMonth()+1).padStart(2,"0")}-${String(t.getDate()).padStart(2,"0")}`;

}


// ── AUTH ──────────────────────────────────────────────────────────────────────

async function signIn() {

  try {

    if (!msalInstance) {

      msalInstance = new msal.PublicClientApplication(MSAL_CONFIG);

      await msalInstance.initialize();

    }

    const result = await msalInstance.loginPopup({

      scopes: ["User.Read", "Files.ReadWrite"]

    });

    currentAccount = result.account;

    showSignedIn(currentAccount);

  } catch(e) {

    showToast("Sign-in failed: " + e.message, "err");

  }

}

function signOut() {

  msalInstance.logoutPopup({ account: currentAccount });

}


function showSignedIn(account) {

  document.getElementById("signInWall").style.display = "none";

  document.getElementById("formWrap").style.display   = "block";


  const name     = account.name || account.username;

  const initials = name.split(" ").map(n => n[0]).join("").substring(0,2).toUpperCase();

  document.getElementById("userInitials").textContent = initials;

  document.getElementById("userName").textContent     = name;

  document.getElementById("submittedBy").value        = account.username;

  document.getElementById("userBar").classList.add("show");

}


// ── GET GRAPH TOKEN ──────────────────────────────────────────────────────────

async function getGraphToken() {

  const request = { scopes: ["Files.ReadWrite"], account: currentAccount };

  try {

    const result = await msalInstance.acquireTokenSilent(request);

    return result.accessToken;

  } catch(e) {

    const result = await msalInstance.acquireTokenPopup(request);

    return result.accessToken;

  }

}


// ── UPLOAD RECEIPT TO ONEDRIVE ────────────────────────────────────────────────

async function uploadToOneDrive(file, dateStr, vendor) {

  const token    = await getGraphToken();

  const safeName = `${dateStr}_${vendor.replace(/[^a-zA-Z0-9]/g,"_").substring(0,25)}_${file.name}`;

  const endpoint = `https://graph.microsoft.com/v1.0/me/drive/root:/${ONEDRIVE_FOLDER}/${safeName}:/content`;


  showProgress(10, "Creating OneDrive folder...");


  // Ensure folder exists

  await fetch(`https://graph.microsoft.com/v1.0/me/drive/root/children`, {

    method: "POST",

    headers: { Authorization: `Bearer ${token}`, "Content-Type": "application/json" },

    body: JSON.stringify({

      name: ONEDRIVE_FOLDER,

      folder: {},

      "@microsoft.graph.conflictBehavior": "rename"

    })

  });


  showProgress(30, "Uploading receipt...");


  const uploadRes = await fetch(endpoint, {

    method: "PUT",

    headers: { Authorization: `Bearer ${token}`, "Content-Type": file.type || "application/octet-stream" },

    body: file

  });


  if (!uploadRes.ok) throw new Error("Upload failed: " + uploadRes.statusText);


  showProgress(75, "Getting shareable link...");

  const fileData = await uploadRes.json();


  // Create sharing link

  const shareRes = await fetch(

    `https://graph.microsoft.com/v1.0/me/drive/items/${fileData.id}/createLink`,

    {

      method: "POST",

      headers: { Authorization: `Bearer ${token}`, "Content-Type": "application/json" },

      body: JSON.stringify({ type: "view", scope: "anonymous" })

    }

  );


  showProgress(95, "Finalizing...");

  const shareData = await shareRes.json();

  return shareData.link?.webUrl || fileData.webUrl;

}

// ── RECEIPT FILE HANDLING ─────────────────────────────────────────────────────

let receiptFile = null;


function handleFile(input) {

  const file = input.files[0];

  if (!file) return;

  receiptFile = file;

  document.getElementById("previewName").textContent = file.name;

  if (file.type.startsWith("image/")) {

    const reader = new FileReader();

    reader.onload = e => {

      document.getElementById("previewImg").src = e.target.result;

      document.getElementById("previewWrap").classList.add("show");

    };

    reader.readAsDataURL(file);

  } else {

    document.getElementById("previewImg").src = "";

    document.getElementById("previewWrap").classList.add("show");

  }

}

  document.getElementById("fileGallery").value = "";

  hideProgress();

}

// ── SUBMIT ────────────────────────────────────────────────────────────────────

async function submitForm() {

  const date     = document.getElementById("date").value;

  const category = document.getElementById("category").value;

  const vendor   = document.getElementById("vendor").value.trim();

  const payment  = document.getElementById("payment").value;

  const amount   = parseFloat(document.getElementById("amount").value) || 0;


  if (!date || !category || !vendor || !payment || amount <= 0) {

    showToast("⚠ Fill in all required fields", "err");

    return;

  }


  const btn   = document.getElementById("submitBtn");

  const label = document.getElementById("submitLabel");

  btn.disabled = true;

  label.innerHTML = '<div class="spinner"></div> Saving...';


  try {

    // 1. Upload receipt to OneDrive (if selected)

    let receiptLink = "";

    if (receiptFile) {

      receiptLink = await uploadToOneDrive(receiptFile, date, vendor);

      showProgress(100, "Upload complete!");

      await new Promise(r => setTimeout(r, 600));

      hideProgress();

    }


    // 2. Send expense data to Apps Script → Google Sheets

    label.innerHTML = '<div class="spinner"></div> Writing to Sheet...';

    const payload = {

      date        : date,

      category    : category,

      subcategory : document.getElementById("subcategory").value,

      vendor      : vendor,

      description : document.getElementById("description").value.trim(),

      payment     : payment,

      amount      : amount.toFixed(2),

      tax         : (parseFloat(document.getElementById("tax").value) || 0).toFixed(2),

      approvedBy  : document.getElementById("approvedBy").value.trim(),

      receiptNo   : document.getElementById("receiptNo").value.trim(),

      submittedBy : document.getElementById("submittedBy").value.trim(),

      receiptLink : receiptLink,

    };


    const res  = await fetch(APPS_SCRIPT_URL, {

      method: "POST",

      body: JSON.stringify(payload),

    });

    const data = await res.json();


    if (data.success) {

      const total = (parseFloat(payload.amount) + parseFloat(payload.tax)).toFixed(2);

      document.getElementById("successMsg").innerHTML =

        `Total $${total} saved to Google Sheets.${receiptLink ? "<br>Receipt uploaded to OneDrive." : ""}`;

      document.getElementById("formWrap").style.display    = "none";

      document.getElementById("successScreen").classList.add("show");

    } else {

      throw new Error(data.error || "Unknown error from server");

    }


  } catch(err) {

    showToast("❌ " + err.message, "err");

  } finally {

    btn.disabled = false;

    label.textContent = "Save Expense";

  }

}


function clearForm() {

  ["vendor","description","approvedBy","receiptNo"].forEach(id => document.getElementById(id).value = "");

  document.getElementById("amount").value   = "";

  document.getElementById("tax").value      = "";

  document.getElementById("payment").value  = "";

  document.getElementById("subcategory").innerHTML = '<option value="">— Select category first —</option>';

  document.getElementById("totalDisplay").textContent = "$0.00";

  removeImage();

  setToday();

}


function resetForm() {

  document.getElementById("successScreen").classList.remove("show");

  document.getElementById("formWrap").style.display = "block";

  clearForm();

}
